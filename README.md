# Java-- Example Application

	mkdir ./jmmtest
	cd ./jmmtest
	jmm here .

	jmm get github.com/javaminusminus/jmmexample
	jmm test ./src/github/com/javaminusminus/jmmexample
	jmm install ./src/github/com/javaminusminus/jmmexample
	jmmexample
